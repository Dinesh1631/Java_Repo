import java.util.*;
class HelloWorld {
    public static void main(String[] args) {
        System.out.println("Main Thread started here");
        
        Thread thread1 = new Thread1();
        //thread1.setDaemon(true);
        thread1.start();
        
        Thread thread2 = new Thread(new Thread2());
        //thread2.setDaemon(true);
        //The setDaemon method used the make the thread daemon.
        thread2.start();
        
        System.out.println("Main Thread ended here");
    }
}
//This belongs to thread1(extends threads class).
class Thread1 extends Thread{
    // @override
    public void run(){
        for(int i =0;i<5;i++)
        {
        System.out.println(Thread.currentThread()+","+i);
        }
    }
}
//This belongs to thread2(Implements runnable interface).
class Thread2 implements Runnable{
    // @override
    public void run(){
        for(int i =0;i<5;i++)
        {
        System.out.println("This belongs to Thread2 "+ i);
        }
    }
}
